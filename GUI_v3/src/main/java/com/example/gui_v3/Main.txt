package com.example.gui_v3;
import javafx.scene.Scene;
import javafx.scene.control.RadioButton;
import javafx.scene.control.Tab;
import javafx.scene.control.TabPane;
import javafx.scene.control.ToggleButton;
import javafx.stage.Stage;
import javafx.scene.text.FontWeight;
import javafx.application.Application;

public abstract class Main extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception{
        Tab tabComboBox = new Tab("комбо коробка", comboBoxExample());
        Tab tabToggleButton = new Tab("toggle button", ToggleButtonExample());
        Tab tabRadioButtons = new Tab ("радио кнопка", RadioButtonExample());
        Tab tabSpinners = new Tab("спинеры", spinnerExample());
        TabPane root = new TabPane();
        root.getTabs().addAll(tabSpinners,tabComboBox,tabRadioButtons, tabToggleButton);
        primaryStage.setTitle("стэйдж");
        primaryStage.setScene(new Scene(root));
        primaryStage.show();
    }
}
